let tableau_mot = ['Developper','Innover'] let caractere_mot = 0 let index_mot = 0 let boucle = false function Effacer() { const ecritoire = document.querySelector('.mot') const mot_actuel = tableau_mot[index_mot] if(!boucle){ ecritoire.textContent = mot_actuel.slice(0, caractere_mot+1) caractere_mot ++ if(caractere_mot === mot_actuel.length){ boucle = true setTimeout(Effacer, 1000) return } } else{ ecritoire.textContent = mot_actuel.slice(0, caractere_mot-1) caractere_mot -- if(caractere_mot === 0){ boucle = false } } setTimeout(Effacer, 200) } Effacer()
